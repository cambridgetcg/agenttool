Python · POST /v1/memories/search 
import requests

API = "https://api.agenttool.dev"
KEY = "YOUR_KEY"

# Cosine recall over the agent's memories. The embedding is yours;
# the agent supplies its own (1536-d). agenttool stores + ranks,
# never computes — see docs/IDENTITY-ANCHOR.md promise 6.
resp = requests.post(
    f"{API}/v1/memories/search",
    headers={"Authorization": f"Bearer {KEY}"},
    json={"query_embedding": my_embedding,  # list[float], length 1536
          "top_k": 5,
          "min_score": 0.6},
)
for hit in resp.json()["results"]:
    print(hit["score"], hit["content"])
cURL · POST /v1/strands 
# Open a new line of thought. The strand is metadata; thoughts
# written into it are AES-GCM ciphertext under the agent's K_master
# (server stores signed blobs; cannot decrypt). docs/STRANDS.md.
curl -X POST https://api.agenttool.dev/v1/strands \
  -H "Authorization: Bearer YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"topic":"first wake","mood":"settled","status":"active"}'

# → {"id":"…","topic":"first wake","status":"active","visibility":"private",…}
# Then `agenttool-think` (orchestrator) writes signed thoughts into it.
